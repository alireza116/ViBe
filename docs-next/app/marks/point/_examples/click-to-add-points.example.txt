mount(Elicit({
  width: 380, height: 260,
  margins: { top: 14, right: 14, bottom: 26, left: 30 },
  data: [
    { x: 25, y: 40, group: "a" },
    { x: 60, y: 70, group: "a" },
  ],
  schema: {
    x: { type: "quantitative", domain: [0, 100] },
    y: { type: "quantitative", domain: [0, 100] },
    group: { type: "categorical", default: "a" },
  },
  marks: [
    point({
      fill: "#0d9488",
      size: 8,
      channels: {
        x: { field: "x", edit: drag() },
        y: { field: "y", edit: drag() },
      },
      // dblclick, not click: a plain click would race the move-drag on the same
      // mark. The pointer inverts through x and y to mint the datum; group comes
      // from the schema default. count caps the dataset.
      edits: [ create({ gesture: "dblclick" }) ],
      constraints: [ count({ max: 10 }) ],
    }),
  ],
}))
