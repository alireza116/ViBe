mount(Elicit({
  width: 420, height: 260,
  margins: { top: 16, right: 16, bottom: 30, left: 40 },
  data: [
    { x: 1, y: 8,  tag: "a" },
    { x: 3, y: 3,  tag: "b" },
    { x: 5, y: 6,  tag: "c" },
    { x: 7, y: 2,  tag: "d" },
  ],
  schema: {
    x:   { type: "quantitative", domain: [0, 8] },
    y:   { type: "quantitative", domain: [0, 10] },
    tag: { type: "categorical", domain: ["a", "b", "c", "d"] },
  },
  marks: [
    // `width` / `height` are pixel extents, so each rect is a fixed 28x28 box
    // centred on its (x, y) anchor — the SVG-native form, alongside x1/x2 spans
    // and band cells. The anchor stays a positional channel: a drag on x and y
    // inverts the pointer through both scales, so the whole box moves.
    rect({
      fill: "#2563eb", fillOpacity: 0.8,
      channels: {
        x:      { field: "x", edit: edit.drag() },
        y:      { field: "y", edit: edit.drag() },
        width:  { value: 28 },
        height: { value: 28 },
      },
    }),
  ],
}))
