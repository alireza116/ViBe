  axes: { x: { title: "when" }, y: { title: "belief" } },

    point({
      size: 7, fill: "#0d9488", stroke: "#0f5c53", strokeWidth: 1,
      channels: {
        x: { field: "when" }, y: { field: "belief" },
      },
      edits: [
        create({ trigger: "dblclick" }),
        drag({ channels: ["x", "y"] }),
      ],
    }),
  ],
