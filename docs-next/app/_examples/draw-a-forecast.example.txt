mount(Elicit({
  width: 560, height: 280,
  responsive: "reflow",
  margins: { top: 24, right: 24, bottom: 48, left: 56 },
  data: [
    { month: 0, demand: 50 }, { month: 1, demand: 50 },
    { month: 2, demand: 50 }, { month: 3, demand: 50 },
    { month: 4, demand: 50 }, { month: 5, demand: 50 },
    { month: 6, demand: 50 }, { month: 7, demand: 50 },
  ],
  schema: {
    month:  { type: "quantitative", domain: [0, 7] },
    demand: { type: "quantitative", domain: [0, 100] },
  },
  axes: { x: { title: "Month" }, y: { title: "Demand" } },
  marks: [
    lineY({
      stroke: "#4f46e5", strokeWidth: 3, curve: "catmullRom",
      channels: {
        x: { field: "month" },
        y: { field: "demand",
             edit: drag({ pick: "sweep", guide: true }) },
      },
    }),
  ],
}))
